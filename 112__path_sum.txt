 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */
bool hasPathSum(struct TreeNode* root, int targetSum) {
    if(root==NULL){
        return false;
    }
    if(root->left== NULL && root->right==NULL){
        if(targetSum==root->val) return true;
        return false;
    }
